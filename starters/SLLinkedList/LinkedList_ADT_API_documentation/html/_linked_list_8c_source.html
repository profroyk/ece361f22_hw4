103void insertNodeInLList(LinkedListPtr_t L, long data, int pos){
104 ListNodePtr_t head = L->head;
105 ListNodePtr_t q, p;
106 ListNodePtr_t newNode = (ListNodePtr_t) malloc(sizeof(ListNode_t));
107
108 int k = 1;
109
110 if (L == NULL) {
111 printf("ERROR: LinkedList ADT(insertNodeInLList): Linked List does not exist\n");
112 return;
113 }
114
115 if(!newNode){
116 printf("ERROR: LinkedList ADT(insertNodeInLList): Could not create new node\n");
117 return;
118 }
119 newNode->data = data;
120 p = head;
121 if ((pos == 1) || (p == NULL)){
122 newNode->next = head;
123 L->head = newNode;
124 L->count++;
125 }
126 else {
127 while((p != NULL) && (k < pos)){
128 k++;
129 q = p;
130 p = p->next;
131 }
132 newNode->next = q->next;
133 q->next = newNode;
134 L->count++;
135 }
136}

167 }

172 p = p->next;
